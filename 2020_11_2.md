```js
  let start = 0;
  for (let i = 1; i < A.length; i++) {
```

解题思路：BFS，利用栈，出栈入栈，每个单词按字母表顺序，每个位置改变一个字母，看在 wordlist 中是否能够找到，如果能，就把新的 word 和 level+1 压入栈中

```js
/**
 * @param {string} beginWord
 * @param {string} endWord
 * @param {string[]} wordList
 * @return {number}
 */
var ladderLength = function (beginWord, endWord, wordList) {
  const wordSet = new Set(wordList);
  let queue = [];
  queue.push([beginWord, 1]);
  while (queue.length) {
    let levelSize = queue.length;
    for (let i = 0; i < levelSize; i++) {
      let [word, level] = queue.shift();
      if (word === endWord) {
        return level;
      }
      for (let i = 0; i < word.length; i++) {
        for (let c = 97; c <= 122; c++) {
          // 对应26个小写字母
          const newWord =
            word.slice(0, i) + String.fromCharCode(c) + word.slice(i + 1); // 形成新词
          if (wordSet.has(newWord)) {
            // 单词表里有这个新词
            queue.push([newWord, level + 1]); // 作为下一层的词入列
            wordSet.delete(newWord); // 避免该词重复入列
          }
        }
      }
    }
  }
  return 0; //当找不到结尾节点的时候，返回零，不一定
};
```

```js
  if (left === right) {
    return 0;
  } else {
    const mid = Math.floor((left + right) / 2);
    const n1 = countRangeSumRecursive(sum, lower, upper, left, mid);
    const n2 = countRangeSumRecursive(sum, lower, upper, mid + 1, right);
    let ret = n1 + n2;

    // 首先统计下标对的数量
    let i = left;
    let l = mid + 1;
    let r = mid + 1;
    while (i <= mid) {
      while (l <= right && sum[l] - sum[i] < lower) l++;
      while (r <= right && sum[r] - sum[i] <= upper) r++;
      ret += r - l;
      i++;
    }

    // 随后合并两个排序数组,归并排序
    const sorted = new Array(right - left + 1);
    let p1 = left,
      p2 = mid + 1;
    let p = 0;
    while (p1 <= mid || p2 <= right) {
      if (p1 > mid) {
        sorted[p++] = sum[p2++];
      } else if (p2 > right) {
        sorted[p++] = sum[p1++];
      } else {
        if (sum[p1] < sum[p2]) {
          sorted[p++] = sum[p1++];
        } else {
          sorted[p++] = sum[p2++];
        }
      }
    }
    for (let i = 0; i < sorted.length; i++) {
      sum[left + i] = sorted[i];
    }
    return ret;
  }
```

```js
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
function nextPermutation(nums) {
  let i = nums.length - 2;
  while (i >= 0 && nums[i] >= nums[i + 1]) i--;
  let j = nums.length - 1;
  if (i >= 0) {
    while (j >= 0 && nums[i] >= nums[j]) {
      j--;
    }
    [nums[i], nums[j]] = [nums[j], nums[i]];
  }

  let l = i + 1;
  let r = nums.length - 1;
  while (l < r) {
    [nums[l], nums[r]] = [nums[r], nums[l]];
    l++;
    r--;
  }
}
```

## 自由之路

```

旋转 ring 拼出 key 字符 key[i] 的阶段中：

您可以将 ring 顺时针或逆时针旋转一个位置，计为1步。旋转的最终目的是将字符串 ring 的一个字符与 12:00 方向对齐，并且这个字符必须等于字符 key[i] 。
如果字符 key[i] 已经对齐到12:00方向，您需要按下中心按钮进行拼写，这也将算作 1 步。按完之后，您可以开始拼写 key 的下一个字符（下一阶段）, 直至完成所有拼写。

来源：力扣（LeetCode）
链接：https://leetcode-cn.com/problems/freedom-trail
著作权归领扣网络所有。商业转载请联系官方授权，非商业转载请注明出处。
```

```
输入: ring = "godding", key = "gd"
输出: 4
解释:
 对于 key 的第一个字符 'g'，已经在正确的位置, 我们只需要1步来拼写这个字符。
 对于 key 的第二个字符 'd'，我们需要逆时针旋转 ring "godding" 2步使它变成 "ddinggo"。
 当然, 我们还需要1步进行拼写。
 因此最终的输出是 4。

```

```js
/**
 * @param {string} ring
 * @param {string} key
 * @return {number}
 */
var findRotateSteps = function (ring, key) {
  let dp = []; //动态规划
  dp[0] = [];
  var res = Infinity;
  for (let i = 0; i < ring.length; i++) {
    //记录初始状态
    if (ring[i] === key[0]) {
      //算顺时针和逆时针的
      dp[0][i] = Math.min(i, ring.length - i) + 1; //加上按下的
    }
  }
  for (let i = 1; i < key.length; i++) {
    dp[i] = [];
    for (let j = 0; j < ring.length; j++) {
      if (ring[j] === key[i]) {
        for (let k = 0; k < ring.length; k++) {
          if (dp[i - 1][k]) {
            let temp =
              dp[i - 1][k] +
              1 +
              Math.min(Math.abs(k - j), ring.length - Math.abs(k - j)); //计算从上一步到下一步的顺逆时针，加上按下按钮的那一步
            dp[i][j] = dp[i][j] ? Math.min(dp[i][j], temp) : temp; //算最终值
          }
        }
      }
    }
  }
  for (let i = 0; i < ring.length; i++) {
    if (dp[key.length - 1][i]) {
      res = Math.min(res, dp[key.length - 1][i]);
    }
  }
  return res;
};
```

```js
var sortArrayByParityII = function (A) {
```

```js
var oddEvenList = function (head) {

  if (!head.next) return head;
```

```js
  let limit = start;
  let rest = gas[start];

    start = start % gas.length; //回到原点，需要除模

      return start;
    }
  }
```

## 对链表进行插入排序

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var insertionSortList = function (head) {
  if (head === null) return head;
  let resHead = new ListNode(0);
  resHead.next = head;
  let curr = head.next;
  let preSorted = head; //固定一个已排好序的的节点，永远都是已排好序的最后一个

  while (curr !== null) {
    if (preSorted.val <= curr.val) {
      preSorted = preSorted.next; //已排好序的节点往右移动
    } else {
      let pre = resHead;
      while (pre.next.val <= curr.val) {
        pre = pre.next;
      }
      //这种需要插入的，已排好序的最后一个节点还是原来那个，但是下一个节点要更新为curr.next
      preSorted.next = curr.next;
      //插入curr
      curr.next = pre.next;
      pre.next = curr;
    }

    curr = preSorted.next;
  }
  return resHead.next;
};
```

## 排序链表

```js
//归并排序链表
/**
 * Definition for singly-linked list.
 * function ListNode(val, next) {
 *     this.val = (val===undefined ? 0 : val)
 *     this.next = (next===undefined ? null : next)
 * }
 */
/**
 * @param {ListNode} head
 * @return {ListNode}
 */
var sortList = function (head) {
  if (!head || !head.next) return head; //限制条件
  var p = [head, head.next];
  while (p[1] && p[1].next) {
    p = [p[0].next, p[1].next.next];
  }
  var mid = p[0].next; //找到中间节点
  p[0].next = null; //让另外一条指针的尾部节点为null,链表的特性
  p = [sortList(head), sortList(mid)]; //sortList是函数
  var h = new ListNode(0);
  var res = h;
  //归并排序的核心思想，两边数组比较，然后拼接
  while (p[0] && p[1]) {
    p[0].val < p[1].val
      ? ((h.next = p[0]), (p[0] = p[0].next))
      : ((h.next = p[1]), (p[1] = p[1].next));
    h = h.next;
  }
  h.next = p[0] || p[1];
  return res.next;
};
```

## 在排序数组中查找元素的第一个和最后一个位置

进阶：

你可以设计并实现时间复杂度为  O(log n)  的算法解决此问题吗？

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var searchRange = function (nums, target) {
  let ans = [-1, -1];
  let leftindex = binarySearch(nums, target, true);
  let rightindex = binarySearch(nums, target, false) - 1;
  if (
    leftindex <= rightindex &&
    rightindex < nums.length &&
    nums[leftindex] === target &&
    nums[rightindex] === target
  ) {
    ans = [leftindex, rightindex];
  }
  return ans;
};

const binarySearch = (nums, target, lower) => {
  let left = 0,
    right = nums.length - 1,
    ans = nums.length;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (nums[mid] > target || (lower && nums[mid] >= target)) {
      //当寻找的是左值，Lower为true
      //当寻找的是右值，lower为false
      right = mid - 1;
      ans = mid; //更新当前的ans为mid
    } else {
      //当找到最后一个等于target的时候，会到这边来
      left = mid + 1;
    }
  }
  return ans;
};
```
