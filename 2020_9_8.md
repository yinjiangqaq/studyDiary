## 分数到小数

给定两个整数，分别表示分数的分子  numerator 和分母 denominator，以字符串形式返回小数。

如果小数部分为循环小数，则将循环的部分括在括号内。

示例：

> 输入: numerator = 2, denominator = 3
>
> 输出: "0.(6)"

```js
/**
 * @param {number} numerator
 * @param {number} denominator
 * @return {string}
 */
var fractionToDecimal = function (numerator, denominator) {
  if (!numerator) return "0";
  let res = [];
  let map = {};
  if ((numerator > 0) ^ (denominator > 0)) res.push("-");
  let num = Math.abs(numerator);
  let deno = Math.abs(denominator);
  res.push(Math.floor(num / deno).toString());
  let remain = num % deno;
  if (remain) res.push(".");
  while (remain) {
    if (map[remain]) {
      //有相同的余数时，确认为循环
      res.splice(map[remain], 0, "(");
      res.push(")");
      break;
    }
    map[remain] = res.length;
    remain *= 10;
    res.push(Math.floor(remain / deno).toString());
    remain %= deno;
  }
  return res.join("");
};
```

## 手写 promise.all

```js
Promise.all = (list) => {
  return new Promise((resolve, reject) => {
    const results = [];
    let count = 0;
    list.forEach((item, index) => {
      item.then(
        (res) => {
          results[index] = res;
          if (++count === list.length) {
            resolve(results);
          }
        },
        (rej) => reject(rej)
      );
    });
  });
};
```

## 手写 promise.race

```js
//promise.race
function isPromise(obj) {
  return !!obj && (typeof obj === "object" || typeof obj === "function");
}

const myPromiseRace = (arr) => {
  return new Promise((resolve, reject) => {
    if (arr.length === 0) {
      return;
    } else {
      for (let item of arr) {
        if (isPromise(item)) {
          item.then(
            (data) => {
              resolve(data);
            },
            (err) => reject(err)
          );
        }
      }
    }
  });
};

var promise1 = new Promise(function (resolve, reject) {
  setTimeout(resolve, 500, "one");
});

var promise2 = new Promise(function (resolve, reject) {
  setTimeout(resolve, 100, "two");
});

myPromiseRace([promise1, promise2]).then(function (value) {
  console.log(value); // two
});
```

## js 函数定义的方式

```js

  return a + b;
}
```
