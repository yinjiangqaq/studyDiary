## 消失的数字

数组 nums 包含从 0 到 n 的所有整数，但其中缺了一个。请编写代码找出那个缺失的整数。你有办法在 O(n)时间内完成吗？

```
示例 1：

输入：[3,0,1]
输出：2


示例 2：

输入：[9,6,4,2,3,5,7,0,1]
输出：8
```

```js
//利用异或的位运算
/**
 * @param {number[]} nums
 * @return {number}
 */
var missingNumber = function (nums) {
  var a = 0;
  for (let i = 1; i <= nums.length; i++) {
    a = a ^ nums[i - 1] ^ i;
  }
  return a;
};
```

## 买卖股票 2

给定一个数组，它的第 i 个元素是一支给定股票第 i 天的价格。

设计一个算法来计算你所能获取的最大利润。你可以尽可能地完成更多的交易（多次买卖一支股票）。

```
示例 1:

输入: [7,1,5,3,6,4]
输出: 7
解释: 在第 2 天（股票价格 = 1）的时候买入，在第 3 天（股票价格 = 5）的时候卖出, 这笔交易所能获得利润 = 5-1 = 4 。
     随后，在第 4 天（股票价格 = 3）的时候买入，在第 5 天（股票价格 = 6）的时候卖出, 这笔交易所能获得利润 = 6-3 = 3 。


```

```js
//找到多个顶点和谷底，相减拿到最高的利润
/**
 * @param {number[]} prices
 * @return {number}
 */
var maxProfit = function (prices) {
  let maxProfit = 0;
  let valley = prices[0];
  let peak = prices[0];
  var i = 0;
  while (i < prices.length - 1) {
    while (i + 1 < prices.length && prices[i] >= prices[i + 1]) i++;
    valley = prices[i];
    while (i + 1 < prices.length && prices[i] <= prices[i + 1]) i++;
    peak = prices[i];
    maxProfit += peak - valley;
  }
  return maxProfit;
};
```

```js
//快慢指针，把两个指针拉到同一个起点开始
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */

/**
 * @param {ListNode} headA
 * @param {ListNode} headB
 * @return {ListNode}
 */
var getIntersectionNode = function (headA, headB) {
  let alength = 0;
  let node = headA;
  while (node) {
    alength++;
    node = node.next;
  }
  if (!alength) return null;
  node = headB;
  let blength = 0;
  while (node) {
    blength++;
    node = node.next;
  }
  if (!blength) return null;
  let slow, fast;
  let diff = 0;
  if (alength > blength) {
    slow = headA;
    fast = headB;
    diff = alength - blength;
  } else {
    slow = headB;
    fast = headA;
    diff = blength - alength;
  }
  while (diff--) {
    slow = slow.next;
  }
  while (slow !== fast) {
    slow = slow.next;
    fast = fast.next;
  }
  return slow;
};
```

```js
function solution(root, k) {
  if (!root) return root;
  let p1 = root;
  let p2 = root;
  for (let i = 0; i < k; i++) {
    p2 = p2.next; //快它k步
  }
  while (p2) {
    p1 = p1.next;
```

## URL 化

URL 化。编写一种方法，将字符串中的空格全部替换为%20。假定该字符串尾部有足够的空间存放新增字符，并且知道字符串的“真实”长度。

```
 输入："Mr John Smith    ", 13
 输出："Mr%20John%20Smith"
```

```js
/**
 * @param {string} S
 * @param {number} length
 * @return {string}
 */
var replaceSpaces = function (S, length) {
  //截取规定长度，然后按照分割重组
  return S.substr(0, length).split(" ").join("%20");
};
```

```
s = "abaccdeff"
返回 "b"

s = ""
返回 " "
```

```js
/**
 * @param {string} s
 * @return {character}
 */
var firstUniqChar = function (s) {
  for (let char of new Set(s)) {
    if (s.match(new RegExp(char, "g")).length === 1) {
      return char;
    }
  }
  return " ";
};
```

## 剑指 Offer 55 - II. 平衡二叉树

输入一棵二叉树的根节点，判断该树是不是平衡二叉树。如果某二叉树中任意节点的左右子树的深度相差不超过 1，那么它就是一棵平衡二叉树。

```
    3
   / \
  9  20
    /  \
   15   7
   true

         1
      / \
     2   2
    / \
   3   3
  / \
 4   4
 false
```

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
//递归的方法是找最大高度的递归法，然后在计算左右两棵子树的高度差的绝对值大于一返回false
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isBalanced = function (root) {
  var res = true;
  var dfs = function (root) {
    if (!root) return 0;

    var left = dfs(root.left) + 1;

    var right = dfs(root.right) + 1;
    if (Math.abs(left - right) > 1) {
      res = false;
    }
    return Math.max(left, right);
  };
  dfs(root);
  return res;
};
```

## 剑指 Offer 28. 对称的二叉树

```
例如，二叉树 [1,2,2,3,4,4,3] 是对称的。

    1
   / \
  2   2
 / \ / \
3  4 4  3
```

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isSymmetric = function (root) {
  var check = function (p, q) {
    if (!q && !p) return true;
    if (!q || !p) return false;
    return q.val === p.val && check(p.left, q.right) && check(p.right, q.left);
  };
  //不要root.left,root.right，因为不知道有没有左右子树
  return check(root, root);
};
```

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} val
 * @return {ListNode}
 */
var deleteNode = function (head, val) {
  //设置一个前置节点
  let pre = new ListNode(-1);
  pre.next = head;
  node = pre;
  while (node.next) {
    if (node.next.val === val) {
      node.next = node.next.next;
      break;
    }
    node = node.next;
  }
  return pre.next;
};
```

```js
/**
 * @param {number} a
 * @param {number} b
 * @return {number}
 */
var add = function (a, b) {
  if (a == 0) return b;

  return add(a ^ b, (a & b) << 1);
```
